/**MODULO SKATE*/

/**Servicio GET Plantilla skate*/
SELECT * FROM skate
/**Servicio POST Plantilla skate*/
BEGIN
INSERT INTO SKATE (ID, BRAND, MODEL, CATEGORY_ID, NAME) VALUES (:id,:brand,:model,:category_id,:name);
 :status_code:=201;
END;
/**Servicio PUT Plantilla skate*/
BEGIN
UPDATE SKATE SET BRAND=:brand, MODEL=:model, CATEGORY_ID=:category_id, NAME=:name WHERE ID=:id;
 :status_code:=201;
END;
/**Servicio DELETE Plantilla skate*/
BEGIN
DELETE FROM SKATE WHERE ID=:id;
 :status_code:=204;
END;
/**Servicio GET Plantilla skate/:id*/
SELECT * FROM SKATE WHERE ID=:id
